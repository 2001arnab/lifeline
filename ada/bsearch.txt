#include <iostream>

using namespace std;

int binary(int arr[], int length, int ele){
    int left = 0;
    int right = length-1;
    while(left<=right){
        int mid = (left+right)/2;   
        if(arr[mid]==ele){
            return mid;
        }
        else if(arr[mid]>ele){
            right=mid-1;
        }
        else{
            left=mid+1;
        }
    }
    return -1;

}

int main(){
    int arr[]={2,4,5,6,8,9,10,12};
    int length=sizeof(arr)/sizeof(arr[0]);
    cout<<"enter number you want to search : ";
    int ele;
    cin>>ele;
    int result=binary(arr, length, ele);
    if (result==-1){
        cout<<"element not found";
    }
    else{
        cout<<"index is : "<<result;
    }
    return 0;
}
